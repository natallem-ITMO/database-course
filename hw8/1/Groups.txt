-- ДЗ-5.2.1 Полную информацию о студентах (StudentId, StudentName,
-- GroupName) С заданным идентификатором (:StudentId);
-- ДЗ-6.2.1 Полная информация о всех студентах (StudentId,
-- StudentName, GroupName)
-- ДЗ-5.8.3 SumMark студентов каждой группы (GroupName)
-- используем hash т.к. у нас не происходит запросов, связанных со
-- сравнением или выборкой id по каким то вычислимым правилам,
-- следовательно необходимости в упорядоченном индексе нет, т.к. он
-- приводит к менее эффективной работе по времени и памяти
-- + индекс на ключ
create unique index groups_group_id_index on Groups using hash (GroupId)

-- ДЗ-6.1.2 Информация о студентах по :GroupName
-- ДЗ-7.1.2 Удаление студентов по :GroupName
-- ДЗ-7.2.4 Перевод всех студентов группы по :*GroupName
-- Реализация на хеше обеспечивает меньшее потребление памяти и более
-- быстрый доступ, т.к. имя может быть большим и высота дерева
-- соотственно увеличится, а т.к. операций по префиксу имени группы
-- нет среди запросов, то использование дерева было бы неоправданно
create index groups_group_name_index on Groups using hash (GroupName)
