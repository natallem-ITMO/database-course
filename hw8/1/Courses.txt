-- ДЗ-5.5.3 ФИО студента и названия предметов которые у него есть,
-- но не 4 или 5
-- используем hash т.к. у нас не происходит запросов, связанных со
-- сравнением или выборкой id по каким то вычислимым правилам,
-- следовательно необходимости в упорядоченном индексе нет, т.к. он
-- приводит к менее эффективной работе по времени и памяти
-- + индекс на ключ
create unique index courses_course_id_index on Courses using hash
    (CourseId)

-- ДЗ-5.3.2 Информацию о студентах с заданной оценкой (:Mark) по
-- предмету С заданным названием (:CourseName);
-- ДЗ-5.4.1 Информацию о студентах не имеющих оценки по :CourseName
-- среди всех
-- ДЗ-5.4.2 Информацию о студентах не имеющих оценки по :CourseName
-- среди тех, у кого он есть
-- Реализация на хеше обеспечивает меньшее потребление памяти и более
-- быстрый доступ, т.к. имя может быть большим и высота дерева
-- соотственно увеличится, а т.к. операций по префиксу имени курса
-- нет среди запросов, то использование дерева было бы неоправданно
create index courses_course_name_index on Courses using hash
    (CourseName)
